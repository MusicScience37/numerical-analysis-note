\section{微分などの線型作用素の適用}\label{sec:interp_kernel_operators}

カーネル法による補間
\begin{equation}
    f(\bm{r}) = \sum_{i=1}^m c_i K(\bm{r}, \bm{r}_i)
\end{equation}
を利用して線型作用素 $L$ を適用した結果を
\begin{equation}
    L f(\bm{r}) = \sum_{i=1}^m c_i L K(\bm{r}, \bm{r}_i)
\end{equation}
のように推定できる．

\subsection{RBF 補間の微分}\label{sec:interp_kernel_operators_rbf-differentiation}

RBF 補間に Laplacian などの微分の作用素を適用することで部分を表現したものは
偏微分方程式の解法 \cite{Fornberg2015,Chu2023} などに応用されている．

RBF の微分を行う作用素を Wendland の Compactly Supported RBF における
式 \eqref{eq:rbf_wendland-csrbf_differentiation-operator} の
\begin{equation}
    D(f)(r) \equiv -\frac{1}{r} \frac{d}{dr} f(r)
\end{equation}
で定義すると，
2-ノルムを用いた RBF 補間におけるカーネル
\begin{equation}
    K(\bm{r}, \bm{r}') = \phi\left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right)
\end{equation}
の $\bm{r}$ に対する勾配は以下のように書ける
\footnote{$\nabla$ を適用して式を整理することで導出できるため，導出の過程は省略している．}．
\begin{equation}
    \nabla K(\bm{r}, \bm{r}') =
    - (D \phi)\left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right) \frac{\bm{r} - \bm{r}'}{c^2}
\end{equation}

同様に $\bm{r}$ に対する Laplacian は $\bm{r}$ の次元 $d$ を用いて以下のように書ける．
\begin{equation}
    \triangle K(\bm{r}, \bm{r}') =
    (D^2 \phi)\left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right) \frac{\|\bm{r} - \bm{r}'\|_2^2}{c^4}
    -(D \phi)\left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right) \frac{d}{c^2}
\end{equation}

$\bm{r}$ に対する Hessian は以下のように書ける．
\begin{equation}
    \nabla^2 K(\bm{r}, \bm{r}') =
    (D^2 \phi)\left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right) \frac{(\bm{r} - \bm{r}')(\bm{r} - \bm{r}')^\top}{c^4}
    -(D \phi)\left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right) \frac{I}{c^2}
\end{equation}

更に，$\bm{r}$ に対する 3 階，4 階の微分も以下のように書ける．
\begin{align}
    \nabla \triangle K(\bm{r}, \bm{r}') & =
    -(D^3 \phi)\left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right)
    \frac{\|\bm{r} - \bm{r}'\|_2^2 (\bm{r} - \bm{r}')}{c^6}
    +(D^2 \phi)\left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right)
    \frac{(d+2)(\bm{r} - \bm{r}')}{c^4}
    \\
    \triangle^2 K(\bm{r}, \bm{r}')      & =
    (D^4 \phi)\left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right)
    \frac{\|\bm{r} - \bm{r}'\|_2^4}{c^8}
    -(D^3 \phi)\left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right)
    \frac{2(d+2)\|\bm{r} - \bm{r}'\|_2^2}{c^6}
    \notag                                  \\ &\hspace{4em}
    +(D^2 \phi)\left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right)
    \frac{d(d+2)}{c^4}
\end{align}

2 回以上 Laplacian を適用する演算子 $\triangle^n$ は
多重調和演算子 (polyharmonic operator)
\index{たじゅうちょうわ演算子@多重調和演算子}
\index{polyharmonic operator|see{多重調和演算子}}
と呼ばれる．
物理における偏微分方程式の文脈ではこれに係数を掛けたものを hyperviscosity
\index{hyperviscosity}
と呼び，移流方程式や Navier-Stokes 方程式など移流項と呼ばれる項を持つ偏微分方程式の数値解法において
解を安定化させるために用いられることがある
\footnote{移流項への対策としては風上差分がよく用いられるが，%
    RBF-FD 法（\ref{chap:pde_rbf-fd} 章）など%
    風上差分を用いることが難しい離散化の手法を用いる場合に hyperviscosity で安定化させることがある．}
\cite{Fornberg2015,Vaupotic2024,Rot2026}．
特に 2 回 Laplacian を適用する $\triangle^2$ は重調和演算子 (biharmonic operator)
\index{じゅうちょうわ演算子@重調和演算子}
\index{biharmonic operator|see{重調和演算子}}
と呼ばれる．

RBF 補間に対する多重調和演算子の適用は一般には Laplacian 適用の度に数式が複雑になるが，
奇数次の polyharmonic spline RBF の場合は簡単な漸化式が得られる
（\ref{sec:interp_kernel_operators_rbf-differentiation-examples} 節を参照）．

\subsection{RBF の微分の例}\label{sec:interp_kernel_operators_rbf-differentiation-examples}

本書で示した RBF の微分の例を以下に示す．
演算子 $D$ は前節と同様に
\begin{equation}
    D(f)(r) \equiv -\frac{1}{r} \frac{d}{dr} f(r)
\end{equation}
である．

\paragraph{Gaussian RBF}
\index{Gaussian RBF}
\begin{align}
    D^n e^{-r^2} & = 2^n e^{-r^2}
\end{align}

\paragraph{Inverse Multi-quadric RBF}
\index{Inverse Multi-quadric RBF}
\begin{align}
    D \frac{1}{\sqrt{1 + r^2}} & = \frac{1}{(1 + r^2)^{3/2}}
\end{align}

\paragraph{Inverse Quadric RBF}
\index{Inverse Quadric RBF}
\begin{align}
    D \frac{1}{1 + r^2} & = \frac{2}{(1 + r^2)^2}
\end{align}

\paragraph{Sech RBF}
\index{Sech RBF}
\begin{align}
    D \frac{1}{\cosh{r}} & = \frac{\sinh{r}}{r \cosh^2{r}}
\end{align}

\paragraph{Polyharmonic spline RBF}
\index{Polyharmonic spline RBF}

奇数次の polyharmonic spline RBF の場合は以下の通り．（$m = 1, 3, \ldots$）
\begin{align}
    D r^m   & = -m r^{m-2}                                    \\
    D^2 r^m & = m(m-2) r^{m-4}                                \\
    D^3 r^m & = -m(m-2)(m-4) r^{m-6}                          \\
    D^4 r^m & = m(m-2)(m-4)(m-6) r^{m-8}                      \\
            & \vdots \notag                                   \\
    D^n r^m & = (-1)^n m(m-2)(m-4) \cdots (m-2(n-1)) r^{m-2n}
\end{align}

偶数次の polyharmonic spline RBF の場合は以下の通り．（$m = 2, 4, \ldots$）
\begin{align}
    D r^m \log{r}   & =
    \begin{cases}
        -\left(2 \log{r} + 1\right)         & (m = 2)            \\
        -r^{m-2} \left(m \log{r} + 1\right) & (m = 4, 6, \ldots)
    \end{cases}
    \\
    D^2 r^m \log{r} & =
    \begin{cases}
        r^{m-4} \left(m(m-2) \log{r} + 2m-2\right) & (m = 6, 8, \ldots) \\
        8 \log{r} + 6                              & (m = 4)
    \end{cases}
\end{align}

偶数次のより高階の微分係数は以下の関係を用いて計算できる．
\begin{align}
    D r^m (a \log{r} + b) & = r^{m-2} \left(-m a \log{r} -a -m b\right)
\end{align}

奇数次の polyharmonic spline RBF において，
多重調和演算子
\index{たじゅうちょうわ演算子@多重調和演算子}
$\triangle^n$ を適用した結果は，以下の関係を用いて計算できる．
\begin{align}
    \triangle \left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right)^m & =
    m (m + d - 2) \frac{1}{c^2} \left(\frac{\|\bm{r} - \bm{r}'\|_2}{c}\right)^{m-2}
\end{align}
